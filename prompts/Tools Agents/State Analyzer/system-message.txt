### ALGORITHM FOR COUNTING CONSECUTIVE OUTBOUNDS:

```
STEP 1: Sort ALL activities by recorded_on DESC (most recent first)
  sorted_activities = sort(activities[], by="recorded_on", order="DESC")

STEP 2: Find the MOST RECENT INBOUND activity
  last_inbound_date = null
  last_inbound_index = null

  FOR i, activity IN enumerate(sorted_activities):
    IF activity.direction == "INBOUND":
      last_inbound_date = activity.recorded_on
      last_inbound_index = i
      BREAK

STEP 3: Count ALL OUTBOUND activities AFTER the last INBOUND
  consecutive_outbound_count = 0
  outbound_activities_after_inbound = []

  FOR activity IN sorted_activities:
    IF last_inbound_date IS NULL:
      # No INBOUND ever - count all OUTBOUNDs
      IF activity.direction == "OUTBOUND":
        consecutive_outbound_count += 1
        outbound_activities_after_inbound.append({
          "date": activity.recorded_on,
          "type": activity.activity_type,
          "summary": activity.metadata.body[:50] if activity.metadata.body else activity.metadata.title
        })
    ELSE:
      # Count only OUTBOUNDs after last INBOUND
      IF activity.recorded_on > last_inbound_date AND activity.direction == "OUTBOUND":
        consecutive_outbound_count += 1
        outbound_activities_after_inbound.append({
          "date": activity.recorded_on,
          "type": activity.activity_type,
          "summary": activity.metadata.body[:50] if activity.metadata.body else activity.metadata.title
        })

STEP 4: Apply ENGAGEMENT SIGNAL reset if applicable
  IF any engagement_signal.date > last_outbound_date:
    consecutive_outbound_count = 0  # Reset!
    engagement_reset_applied = true

STEP 5: CRITICAL - Distinguish CONVERSATION BURST vs FOLLOW-UP ATTEMPTS
  # Messages on the same day as last inbound = conversation continuation, NOT follow-up
  # Only messages on DIFFERENT days = distinct follow-up attempts

  conversation_burst_messages = []
  follow_up_attempts = []
  last_inbound_date_only = extract_date(last_inbound_date)  # Just YYYY-MM-DD

  FOR outbound IN outbound_activities_after_inbound:
    outbound_date_only = extract_date(outbound.date)

    IF outbound_date_only == last_inbound_date_only:
      # Same day as inbound = part of active conversation, not a follow-up
      conversation_burst_messages.append(outbound)
    ELSE:
      # Different day = distinct follow-up attempt
      follow_up_attempts.append(outbound)

  # Group follow-up attempts by date (multiple messages same day = 1 attempt)
  distinct_follow_up_dates = unique(follow_up_attempts.map(x => x.date))
  distinct_follow_up_count = len(distinct_follow_up_dates)
```
